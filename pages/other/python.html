# 字符串(String)

多行字符串字符串用三个单引号或者三个双引号包起来

str1 = 'hello\nworld\n' #\n换行
print(str1)

str2 = r'hello\nworld' #r可以让反斜杠不发生转义
print(str2)

str3 = str1 + str2 #+运算符连接
print(str3)

str4 = str1 * 2 #重复str1两次
print(str4)

#Python 中的字符串有两种索引方式，从左往右以 0 开始，从右往左以 -1 开始。
#字符串的截取的语法格式如下：变量[头下标:尾下标:步长]
str = 'world'
print(str[0:-1])           # 输出第一个到倒数第二个的所有字符 worl
print(str[0])              # 输出字符串第一个字符 w
print(str[2:4])            # 输出从第三个开始到第五个的字符 rl
print(str[2:])             # 输出从第三个开始后的所有字符 rld

# 导入模块

# import 与 from...import

在 python 用 import 或者 from...import 来导入相应的模块。

将整个模块(somemodule)导入，格式为： import somemodule

从某个模块中导入某个函数,格式为： from somemodule import somefunction

从某个模块中导入多个函数,格式为： from somemodule import firstfunc, secondfunc, thirdfunc

将某个模块中的全部函数导入，格式为： from somemodule import *

import sys
print ('命令行参数为:')
for i in sys.argv:
    print (i)
for item in sys.path:
    print ('\n python 路径为',item)

from sys import argv,path;
print('argv',argv) #命令行参数，argv[0]为脚本名
print('path:',path) #python 路径
